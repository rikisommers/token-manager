    ```ts
    const handleGroupsReordered = useCallback(async (newGroupsFromTree: TokenGroup[]) => {
      // --- Snapshot for undo ---
      undoStackRef.current = [
        masterGroups,
        ...undoStackRef.current.slice(0, MAX_UNDO - 1),
      ];

      // --- Apply full cascade including theme sync ---
      // applyGroupMove was already called inside TokenGroupTree's handleDragEnd
      // with (groups, activeId, overId) — result is newGroupsFromTree.
      // Now we need to re-run with themes to get theme sync.
      // We don't have activeId/overId at this point, so we use the simpler approach:
      // the groups are already reordered; sync themes to match master order.
      const newGroups = newGroupsFromTree;

      // Sync theme snapshots: apply the same structural changes to each theme's tokens.
      // For sibling reorder: reorder children arrays to match master.
      // For reparent: applyGroupMove was called in the util — here we apply to themes.
      // Strategy: re-run applyGroupMove on themes using the diff between masterGroups and newGroups.
      //
      // IMPORTANT ARCHITECTURE NOTE:
      // TokenGroupTree calls applyGroupMove(groups, activeId, overId) WITHOUT themes.
      // To sync themes, we need to do a structural sync here.
      //
      // Use syncThemeTokens helper: for each theme, apply the reorder by matching group IDs
      // and rebuilding in the new master order.
      const updatedThemes = themes.map(theme => {
        if (theme.id === '__default__') return theme; // default theme is always rebuilt from masterGroups
        return {
          ...theme,
          tokens: syncThemeOrder(theme.tokens, newGroups),
        };
      });

      // --- Update React state ---
      setMasterGroups(newGroups);
      setThemes(updatedThemes);

      // --- Persist to MongoDB (debounced 300ms) ---
      if (groupReorderSaveTimerRef.current) clearTimeout(groupReorderSaveTimerRef.current);
      groupReorderSaveTimerRef.current = setTimeout(async () => {
        try {
          const rawTokens = tokenService.generateStyleDictionaryOutput(newGroups, globalNamespace);
          const themesPayload = updatedThemes.filter(t => t.id !== '__default__');
          await fetch(`/api/collections/${id}`, {
            method: 'PUT',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ tokens: rawTokens, themes: themesPayload }),
          });
          rawCollectionTokensRef.current = rawTokens as Record<string, unknown>;
          setRawCollectionTokens(rawTokens as Record<string, unknown>);
        } catch {
          // Silent — mirrors existing auto-save error handling pattern
        }
      }, 300);
    }, [masterGroups, themes, id, globalNamespace]);
    ```

    **Step 5 — Implement `syncThemeOrder` pure helper:**
    Add this as a module-level pure function (outside the component, near `updateGroupToken`):

    ```ts
    /**
     * Sync a theme's token groups array to match the master order.
     * Recursively reorders children arrays to match master.
     * Groups present in theme but not master are dropped; groups in master but not theme are skipped.
     */
    function syncThemeOrder(themeGroups: TokenGroup[], masterGroups: TokenGroup[]): TokenGroup[] {
      const themeMap = new Map(themeGroups.map(g => [g.id, g]));
      const result: TokenGroup[] = [];
      for (const masterGroup of masterGroups) {
        const themeGroup = themeMap.get(masterGroup.id);
        if (!themeGroup) continue;
        result.push({
          ...themeGroup,
          children: masterGroup.children && themeGroup.children
            ? syncThemeOrder(themeGroup.children, masterGroup.children)
            : themeGroup.children,
        });
      }
      return result;
    }
    ```
