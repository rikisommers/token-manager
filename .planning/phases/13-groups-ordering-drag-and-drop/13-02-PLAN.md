    ```ts
    }
    ```

    **`handleDragEnd`:**
    ```ts
    function handleDragEnd({ active, over }: DragEndEvent) {
      setActiveNode(null);
      if (!over || active.id === over.id) return;
      const { groups: newGroups } = applyGroupMove(groups, String(active.id), String(over.id));
      onGroupsReordered?.(newGroups);
    }
    ```

    Note: `applyGroupMove` is called without `themes` here ŌĆö theme sync is the page's responsibility (Plan 03 handles it via the `onGroupsReordered` callback, which also triggers theme persistence).

    **JSX structure:**
    Replace the existing `{nodes.map(...)}` render block with:

    ```tsx
    <DndContext
      sensors={sensors}
      collisionDetection={closestCenter}
      onDragStart={handleDragStart}
      onDragEnd={handleDragEnd}
    >
      <SortableContext items={sortedIds} strategy={verticalListSortingStrategy}>
        {flatNodes.map(node => (
          <SortableGroupRow
            key={node.group.id}
            node={node}
            isSelected={node.group.id === selectedGroupId}
            onSelect={id => onGroupSelect?.(id)}
            onDeleteGroup={onDeleteGroup}
            onAddSubGroup={onAddSubGroup}
          />
        ))}
      </SortableContext>
      <DragOverlay>
        {activeNode ? (
          <SortableGroupRow
            node={activeNode}
            isSelected={false}
            onSelect={() => {}}
            isDragOverlay
          />
        ) : null}
      </DragOverlay>
    </DndContext>
    ```

    **The scrollable list container** (`<div className="flex-1 overflow-y-auto py-1">`) wraps the `DndContext` block. The `DragOverlay` renders at document body level (portal) so overflow-y-auto does NOT clip the ghost.
