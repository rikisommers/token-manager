**Step 3 — Create src/services/style-dictionary.service.ts:**

```typescript
/**
 * Style Dictionary Build Service — server-side only.
 * Accepts raw token JSON and returns formatted output strings for all 6 formats.
 * Does NOT write to disk — all output captured in-memory.
 */

import StyleDictionary from 'style-dictionary';
import type { BuildTokensRequest, BuildTokensResult, FormatOutput, BrandFormatOutput } from '@/types';

const FORMATS = ['css', 'scss', 'less', 'js', 'ts', 'json'] as const;
type Format = typeof FORMATS[number];

/**
 * Detect brand keys from token structure.
 *
 * Structure A (flat): { colors: {...}, spacing: {...} } → single brand "globals"
 * Structure B (multi-brand): { brand1: { colors: {...} }, brand2: {...}, globals: {...} }
 *   Brand keys = top-level keys whose values are objects (not token leaves).
 *   A token leaf has both `$value`/`value` and `$type`/`type` keys.
 *
 * For MongoDB tokens stored as { "globals/color-base.json": {...}, "brands/brand1.json": {...} }
 * (file-path keyed), group by first path segment as brand.
 */
function detectBrands(tokens: Record<string, unknown>): { brand: string; tokens: Record<string, unknown> }[] {
  // Check if keys look like file paths (contain '/')
  const keys = Object.keys(tokens);
  const isFilePath = keys.some(k => k.includes('/'));

  if (isFilePath) {
    // File-path keyed: group by first segment
    const brandMap = new Map<string, Record<string, unknown>>();
    for (const [filePath, data] of Object.entries(tokens)) {
      const brand = filePath.split('/')[0];
      if (!brandMap.has(brand)) brandMap.set(brand, {});
      // Merge file data under brand
      const existing = brandMap.get(brand)!;
      if (data && typeof data === 'object' && !Array.isArray(data)) {
        Object.assign(existing, data as Record<string, unknown>);
      }
    }
    return Array.from(brandMap.entries()).map(([brand, t]) => ({ brand, tokens: t }));
  }

  // Flat structure: single brand
  return [{ brand: 'globals', tokens }];
}

/**
 * Build a single brand's tokens into all 6 formats using style-dictionary v5 programmatic API.
 * Returns a map from format name to file content string.
 */
async function buildBrandTokens(
  brandTokens: Record<string, unknown>,
  namespace: string,
  brand: string
): Promise<Map<Format, string>> {
  const results = new Map<Format, string>();

  // Normalize token keys: style-dictionary v5 uses $value/$type
  // Our tokens use `value`/`type` — apply transform to convert
  function normalizeTokens(obj: Record<string, unknown>): Record<string, unknown> {
    const result: Record<string, unknown> = {};
    for (const [key, val] of Object.entries(obj)) {
      if (key === 'value') { result['$value'] = val; }
      else if (key === 'type') { result['$type'] = val; }
      else if (val && typeof val === 'object' && !Array.isArray(val)) {
        result[key] = normalizeTokens(val as Record<string, unknown>);
      } else {
        result[key] = val;
      }
    }
    return result;
  }

  const normalizedTokens = normalizeTokens(brandTokens);

  // Wrap in namespace: { [namespace]: { [brand]: tokens } }
  // Variable names: --{namespace}-{path} e.g. --token-colors-primary
  const wrappedTokens = {
    [namespace || 'token']: normalizedTokens,
  };

  const formatConfigs: Record<Format, { formatter: string; extension: string }> = {
    css: { formatter: 'css/variables', extension: 'css' },
    scss: { formatter: 'scss/variables', extension: 'scss' },
    less: { formatter: 'less/variables', extension: 'less' },
    js: { formatter: 'javascript/es6', extension: 'js' },
    ts: { formatter: 'typescript/es6-declarations', extension: 'd.ts' },
    json: { formatter: 'json/nested', extension: 'json' },
  };

  for (const [fmt, { formatter, extension }] of Object.entries(formatConfigs) as [Format, { formatter: string; extension: string }][]) {
    try {
      const sd = new StyleDictionary({
        tokens: wrappedTokens,
        platforms: {
          [fmt]: {
            transformGroup: fmt === 'css' || fmt === 'scss' || fmt === 'less' ? 'css' : 'js',
            prefix: namespace || 'token',
            files: [
              {
                destination: `tokens-${brand}.${extension}`,
                format: formatter,
                options: {
                  outputReferences: false,
                },
              },
            ],
          },
        },
      });

      // style-dictionary v5: buildAllPlatforms() with file interception
      // Use the 'log' hook to capture output rather than writing to disk
      // Since SD v5 writes to disk, we use a custom file hook approach:
      // Override the file writer by using an in-memory file system via hooks
      let capturedContent = '';

      // SD v5 supports hooks.fileHeader etc., but not in-memory file capture directly.
      // Alternative: configure destination to a temp path and read back, OR
      // Use SD v5's extend + buildPlatform with a custom file writer.
      //
      // Practical approach for server API: use the `format` function directly
      // by accessing sd.getPlatform() and calling the formatter manually.
      //
      // SD v5 exposes: sd.exportPlatform(platformName) which returns formatted file objects.
      // Use this to get content without writing to disk.
      const platformFiles = await sd.exportPlatform(fmt);
      if (platformFiles && platformFiles.length > 0) {
        capturedContent = platformFiles[0].output ?? '';
      }

      results.set(fmt, capturedContent);
    } catch (err) {
      // Format failed — store empty string with error comment
      results.set(fmt, `/* Build error for ${fmt}: ${err instanceof Error ? err.message : String(err)} */`);
    }
  }

  return results;
}

/**
 * Main export: build all formats for all brands from raw token JSON.
 */
export async function buildTokens(request: BuildTokensRequest): Promise<BuildTokensResult> {
  const { tokens, namespace, collectionName } = request;

  const brands = detectBrands(tokens);
  const formatOutputs = new Map<Format, FormatOutput>();

  // Initialize format outputs
  for (const fmt of FORMATS) {
    formatOutputs.set(fmt, { format: fmt, outputs: [] });
  }

  // Build each brand
  for (const { brand, tokens: brandTokens } of brands) {
    const builtFormats = await buildBrandTokens(brandTokens, namespace, brand);

    for (const fmt of FORMATS) {
      const content = builtFormats.get(fmt) ?? '';
      const ext = fmt === 'ts' ? 'd.ts' : fmt;
      const filename = `tokens-${brand}.${ext}`;
      formatOutputs.get(fmt)!.outputs.push({ brand, content, filename });
    }
  }

  return {
    formats: Array.from(formatOutputs.values()),
    collectionName,
  };
}
```

**Step 4 — Create src/app/api/build-tokens/route.ts:**

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { buildTokens } from '@/services/style-dictionary.service';
import type { BuildTokensRequest } from '@/types';

export async function POST(req: NextRequest): Promise<NextResponse> {
  try {
    const body = (await req.json()) as BuildTokensRequest;

    if (!body.tokens || typeof body.tokens !== 'object') {
      return NextResponse.json({ error: 'tokens is required and must be an object' }, { status: 400 });
    }
    if (!body.collectionName || typeof body.collectionName !== 'string') {
      return NextResponse.json({ error: 'collectionName is required' }, { status: 400 });
    }

    const result = await buildTokens({
      tokens: body.tokens,
      namespace: body.namespace ?? 'token',
      collectionName: body.collectionName,
    });

    return NextResponse.json(result);
  } catch (err) {
    console.error('Build tokens error:', err);
    return NextResponse.json(
      { error: err instanceof Error ? err.message : 'Failed to build tokens' },
      { status: 500 }
    );
  }
}
```

**Implementation notes:**
- style-dictionary v5 uses `$value`/`$type` keys (W3C DTCG spec). Our stored tokens use `value`/`type`. The normalizeTokens() function converts before passing to SD.
- `sd.exportPlatform(platformName)` is the SD v5 method for in-memory output — returns array of `{ destination, output }` objects. No disk writes.
- If `exportPlatform` isn't available on the SD v5 instance (API may differ), fall back to `buildAllPlatforms()` with a custom file writer using `hooks` or by overriding the logger. Check SD v5 docs and adapt as needed.
- Variable naming: wrapping tokens under `[namespace]` produces `--token-colors-primary` per spec: `--{namespace}-{path}` (e.g. namespace="token", path="colors.primary").
- The `css` transformGroup applies camelCase-to-kebab-case and adds `--` prefix for CSS custom properties.

```typescript
import JSZip from 'jszip';

async function handleDownloadAll(formats: FormatOutput[], collectionName: string) {
  const zip = new JSZip();
  for (const fmt of formats) {
    for (const { filename, content } of fmt.outputs) {
      zip.file(filename, content);
    }
  }
  const blob = await zip.generateAsync({ type: 'blob' });
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `${collectionName}-tokens.zip`;
  a.click();
  URL.revokeObjectURL(url);
}
```
