Create `src/context/PermissionsContext.tsx` ‚Äî a minimal scaffold that Phase 19 will expand without layout changes.

```typescript
'use client';

import { createContext, useContext } from 'react';
import { useSession } from 'next-auth/react';
import { canPerform } from '@/lib/auth/permissions';
import type { Role, ActionType } from '@/lib/auth/permissions';

interface PermissionsContextValue {
  role: Role | null;
  canPerform: (action: ActionType) => boolean;
}

const PermissionsContext = createContext<PermissionsContextValue>({
  role: null,
  canPerform: () => false, // safe default: deny all when no session
});

export function PermissionsProvider({ children }: { children: React.ReactNode }) {
  const { data: session } = useSession();
  const role = (session?.user?.role as Role) ?? null;

  return (
    <PermissionsContext.Provider
      value={{
        role,
        canPerform: (action) => (role ? canPerform(role, action) : false),
      }}
    >
      {children}
    </PermissionsContext.Provider>
  );
}

export function usePermissions(): PermissionsContextValue {
  return useContext(PermissionsContext);
}
```

Key design points:
- `'use client'` required because it uses `useSession()` (a React hook from next-auth/react)
- Default context value `canPerform: () => false` ensures no accidental permission grants when called outside the provider tree
- `role` cast: `session?.user?.role` is typed as `string | undefined` (from Phase 16 next-auth.d.ts augmentation); cast to `Role | null` is safe because SUPER_ADMIN_EMAIL enforcement in jwt callback ensures only valid Role strings enter the token
- Phase 19 will replace the `canPerform` impl with a version that fetches per-collection overrides ‚Äî the hook API surface (`usePermissions()` returning `{ role, canPerform }`) stays the same
