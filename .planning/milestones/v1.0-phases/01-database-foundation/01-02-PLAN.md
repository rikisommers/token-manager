    ```typescript
    /**
     * Source metadata for tokens imported from GitHub.
     * All fields nullable — tokens created manually have no source.
     */
    export interface ISourceMetadata {
      repo: string | null;    // e.g. "org/design-tokens"
      branch: string | null;  // e.g. "main"
      path: string | null;    // e.g. "tokens/globals"
    }

    /**
     * Plain data shape for a token collection (use for API responses).
     */
    export interface ITokenCollection {
      _id: string;            // MongoDB ObjectId as string
      name: string;           // User-defined collection name (free text)
      tokens: Record<string, unknown>;  // Raw token JSON — W3C Design Token spec object
      sourceMetadata: ISourceMetadata | null;  // GitHub provenance, nullable
      userId: string | null;  // Reserved for multi-user; null in v1
      createdAt: Date;
      updatedAt: Date;
    }

    /**
     * Shape for creating a new collection (omit auto-generated fields).
     */
    export type CreateTokenCollectionInput = Omit<ITokenCollection, '_id' | 'createdAt' | 'updatedAt'>;

    /**
     * Shape for updating an existing collection.
     */
    export type UpdateTokenCollectionInput = Partial<Pick<ITokenCollection, 'name' | 'tokens' | 'sourceMetadata'>>;
    ```

    Note: `tokens` is typed as `Record<string, unknown>` because the W3C Design Token format is a deeply-nested JSON object with arbitrary token names. Mongoose stores it as Mixed type. Do NOT use `any` — use `Record<string, unknown>` or a more specific union if the format is known at planning time.

    Requirements:
    - Use Mongoose with TypeScript generics for the model type.
    - The schema must include all fields from DB-03: `name`, `tokens`, `sourceMetadata`, `userId`, `createdAt`, `updatedAt`.
    - Use `{ timestamps: true }` on the schema to auto-manage `createdAt` and `updatedAt`.
    - Guard against the "next.js hot-reload re-registration" problem: check `mongoose.models.TokenCollection` before calling `mongoose.model(...)`.

    ```typescript
    import mongoose, { Schema, Model } from 'mongoose';
    import type { ITokenCollection } from '@/types/collection.types';

    // Mongoose document type (ITokenCollection without _id string; mongoose adds its own)
    type TokenCollectionDoc = Omit<ITokenCollection, '_id'>;

    const sourceMetadataSchema = new Schema(
      {
        repo:   { type: String, default: null },
        branch: { type: String, default: null },
        path:   { type: String, default: null },
      },
      { _id: false }
    );

    const tokenCollectionSchema = new Schema<TokenCollectionDoc>(
      {
        name:           { type: String, required: true, trim: true },
        tokens:         { type: Schema.Types.Mixed, required: true },
        sourceMetadata: { type: sourceMetadataSchema, default: null },
        userId:         { type: String, default: null, index: true },
      },
      {
        timestamps: true,  // auto createdAt / updatedAt
      }
    );

    // Index for fast listing by name
    tokenCollectionSchema.index({ name: 1 });

    // Guard against Next.js hot-reload model re-registration
    const TokenCollection: Model<TokenCollectionDoc> =
      (mongoose.models.TokenCollection as Model<TokenCollectionDoc>) ||
      mongoose.model<TokenCollectionDoc>('TokenCollection', tokenCollectionSchema);

    export default TokenCollection;
    ```

    Schema design notes:
    - `tokens` uses `Schema.Types.Mixed` because it stores an arbitrary JSON object (W3C token tree). Mongoose will not try to validate its internal structure.
    - `sourceMetadata` uses a nested sub-schema (not a subdocument with its own `_id`).
    - `userId` has a sparse index via `index: true` — this will be used for filtering in Phase 4 multi-user prep.
    - Do NOT add a unique index on `name` — users may have duplicate names across users in future (userId scopes uniqueness).
