**Route 1 — GET /api/figma/test:**

File: `src/app/api/figma/test/route.ts`

Accepts query params: `token` (Figma PAT).

```typescript
import { NextRequest, NextResponse } from 'next/server';

export async function GET(request: NextRequest) {
  const token = request.nextUrl.searchParams.get('token');
  if (!token) {
    return NextResponse.json({ error: 'token parameter required' }, { status: 400 });
  }
  try {
    const response = await fetch('https://api.figma.com/v1/me', {
      headers: { 'X-Figma-Token': token },
    });
    if (!response.ok) {
      return NextResponse.json({ error: 'Invalid Figma token' }, { status: 401 });
    }
    const data = await response.json();
    return NextResponse.json({ ok: true, email: data.email });
  } catch {
    return NextResponse.json({ error: 'Failed to reach Figma API' }, { status: 500 });
  }
}
```

**Implementation steps:**

1. Validate required fields; return 400 if any missing.

2. Fetch variables from Figma:
```typescript
const figmaRes = await fetch(`https://api.figma.com/v1/files/${fileKey}/variables/local`, {
  headers: { 'X-Figma-Token': token },
});
if (!figmaRes.ok) return NextResponse.json({ error: 'Figma API error' }, { status: 502 });
const figmaData = await figmaRes.json();
```

3. Extract the target collection from `figmaData.meta.variableCollections[collectionId]` and all its variables (filter `figmaData.meta.variables` where `variableCollectionId === collectionId`).

4. **Convert to multi-brand token structure** — each mode becomes a top-level brand key:
```typescript
// Result shape: { [modeName]: { [variablePath]: { $value: any, $type: string } } }
const tokensByMode: Record<string, Record<string, unknown>> = {};
const collection = figmaData.meta.variableCollections[collectionId];
for (const mode of collection.modes) {
  const modeTokens: Record<string, unknown> = {};
  for (const variable of variables) {
    const modeValue = variable.valuesByMode[mode.modeId];
    if (modeValue === undefined) continue;
    // Build nested path from variable.name (slash-separated)
    const parts = variable.name.split('/');
    let current: Record<string, unknown> = modeTokens;
    for (let i = 0; i < parts.length - 1; i++) {
      if (!current[parts[i]]) current[parts[i]] = {};
      current = current[parts[i]] as Record<string, unknown>;
    }
    const leafName = parts[parts.length - 1];
    current[leafName] = {
      $value: convertFigmaValue(modeValue, variable.resolvedType),
      $type: mapFigmaType(variable.resolvedType),
    };
  }
  tokensByMode[mode.name] = modeTokens;
}
```

Define `convertFigmaValue` (COLOR: {r,g,b,a} → hex or rgba; FLOAT: numeric; STRING/BOOLEAN: as-is) and `mapFigmaType` (COLOR→'color', FLOAT→'number', STRING→'string', BOOLEAN→'string') as module-level functions in the same file.

5. Save to MongoDB as a new collection:
```typescript
import { dbConnect } from '@/lib/mongodb';

    repo: null,

    path: null,

    figmaFileKey: fileKey,
    figmaCollectionId: collectionId,
  },
  userId: null,
});
return NextResponse.json(
  { collection: { _id: saved._id.toString(), name: saved.name } },

```
