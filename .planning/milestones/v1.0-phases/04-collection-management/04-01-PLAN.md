export async function DELETE(
  _request: Request,
  { params }: { params: { id: string } }
) {
  try {
    await dbConnect();

    const doc = await TokenCollection.findByIdAndDelete(params.id).lean();

    if (!doc) {
      return NextResponse.json({ error: 'Collection not found' }, { status: 404 });
    }

    return NextResponse.json({ success: true });
  } catch (error) {
    console.error('[DELETE /api/collections/[id]]', error);
    return NextResponse.json({ error: 'Failed to delete collection' }, { status: 500 });
  }
}

Create `src/components/CollectionActions.tsx` as a named export `CollectionActions`. This is a self-contained component that owns all three action modals internally and communicates results upward via callbacks.

**Props interface:**

```typescript
interface CollectionActionsProps {
  selectedId: string;            // MongoDB _id of the active collection ('' or 'local' = hidden)
  selectedName: string;          // Current collection name (for modal pre-fills and confirmation text)
  collections: { _id: string; name: string }[];  // Full list for rename inline validation
  onDeleted: () => void;         // Called after successful delete
  onRenamed: (newName: string) => void;   // Called after successful rename with new name
  onDuplicated: (newId: string, newName: string) => void;  // Called after successful duplicate
  onError: (message: string) => void;    // Called on API error for parent to show toast
}
```

**Visibility rule:**
Render `null` (return null early) when `selectedId` is falsy, equals `'local'`, or `collections` is empty. This keeps the component invisible for Local Files.

**Layout:**
Below the selector: a row of three small buttons with icon + label text. Use Tailwind classes matching the existing component style in the codebase (see SaveCollectionDialog.tsx for reference — white bg, border, rounded-md, text-sm, hover states). Spacing: `mt-3 flex gap-2`.

Button styles:
- Delete: `text-red-600 border-red-300 hover:bg-red-50`
- Rename: default (text-gray-700 border-gray-300 hover:bg-gray-50)
- Duplicate: default (text-gray-700 border-gray-300 hover:bg-gray-50)

Button labels (icon + text): "🗑️ Delete", "✏️ Rename", "📋 Duplicate"

**Three modals — all use the same fixed overlay pattern as SaveCollectionDialog.tsx (`fixed inset-0 z-50 flex items-center justify-center bg-black bg-opacity-50`):**

---

**Delete modal (`showDeleteModal` state):**
- Title: "Delete Collection"
- Body: `<p>Delete <strong>"{selectedName}"</strong>? This cannot be undone.</p>`
- Footer buttons: Cancel (gray) + Delete (red: `bg-red-600 hover:bg-red-700 text-white`)
- On Delete click: call `DELETE /api/collections/[selectedId]`; on 200 call `onDeleted()`; on error call `onError('Failed to delete collection')`
- Disable buttons while request is in-flight (`isDeleting` state)

---

**Rename modal (`showRenameModal` state):**
- Title: "Rename Collection"
- Body: text input pre-filled with `selectedName` (reset each time modal opens via useEffect on `showRenameModal`)
- Inline validation: if the trimmed input value matches any `collections[i].name` where `collections[i]._id !== selectedId`, show error text below the input: `"A collection named "{value}" already exists."` and disable the Save button
- Footer: Cancel + Save (blue `bg-blue-600`, disabled when name is blank, unchanged from current, or duplicate)
- On Save: call `PUT /api/collections/[selectedId]` with body `{ name: newName }`; on 200 call `onRenamed(newName)`; on error call `onError('Failed to rename collection')`
- Disable buttons while request is in-flight (`isRenaming` state)

---

**Duplicate modal (`showDuplicateModal` state):**
- Title: "Duplicate Collection"
- Body: text input pre-filled with `"Copy of " + selectedName` (reset each time modal opens via useEffect on `showDuplicateModal`)
- No inline validation needed (duplicate names are handled server-side via POST 409, but for UX simplicity just check blank)
- Footer: Cancel + Duplicate (blue `bg-blue-600`, disabled when name is blank)
- On Duplicate: call `POST /api/collections` with body `{ name: newName, tokens: <fetched from GET /api/collections/[selectedId]>.tokens, sourceMetadata: <same>.sourceMetadata }`.
  - First: fetch `GET /api/collections/[selectedId]` to get the full document (tokens + sourceMetadata).
  - Then: POST to `/api/collections` with the new name and copied data.
  - On 201: call `onDuplicated(response.collection._id, response.collection.name)`
  - On 409 (name conflict): show inline error in the modal: `"A collection named "{name}" already exists."`
  - On other error: call `onError('Failed to duplicate collection')`
- Disable buttons while request is in-flight (`isDuplicating` state)
