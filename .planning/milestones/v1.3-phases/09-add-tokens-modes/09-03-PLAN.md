**Step 2 — Fetch themes on mount:**
Inside `loadCollection()` (the existing async function that fetches the collection), add a parallel fetch for themes:
```ts
const themesRes = await fetch(`/api/collections/${id}/themes`, {
  signal: abortControllerRef.current?.signal,
});
if (themesRes.ok) {
  const themesData = await themesRes.json();
  setThemes(themesData.themes ?? []);
}
```
Add this after the existing `setGraphStateMap` / `graphStateMapRef.current` lines in `loadCollection`.

**Step 3 — Compute filtered groups:**
Add a `useMemo` below the `allFlatGroups` memo:
```ts
const filteredGroups = useMemo(() => {
  if (!activeThemeId) return masterGroups;
  const activeTheme = themes.find(t => t.id === activeThemeId);
  if (!activeTheme) return masterGroups;
  // Recursively filter: keep group if its state is 'enabled' or 'source', remove if 'disabled'
  function filterGroups(groups: TokenGroup[]): TokenGroup[] {
    return groups
      .filter(g => {
        const state = activeTheme.groups[g.id] ?? 'disabled';
        return state !== 'disabled';
      })
      .map(g => ({
        ...g,
        children: g.children ? filterGroups(g.children) : undefined,
      }));
  }
  return filterGroups(masterGroups);
}, [masterGroups, activeThemeId, themes]);
```

**Step 4 — Add theme selector to the actions bar:**
In the JSX actions bar (the `<div className="flex items-center gap-2 border-b ...">` at the top), add the theme selector between the Namespace input and the flex-1 spacer:

```tsx
{/* Theme selector */}
{themes.length > 0 && (
  <div className="flex items-center gap-2">
    <label className="text-sm text-gray-600 whitespace-nowrap">Theme:</label>
    <Select
      value={activeThemeId ?? '__all__'}
      onValueChange={(v) => setActiveThemeId(v === '__all__' ? null : v)}
    >
      <SelectTrigger className="w-36 h-8 text-sm">
        <SelectValue placeholder="All groups" />
      </SelectTrigger>
      <SelectContent>
        <SelectItem value="__all__">All groups</SelectItem>
        {themes.map(t => (
          <SelectItem key={t.id} value={t.id}>{t.name}</SelectItem>
        ))}
      </SelectContent>
    </Select>
  </div>
)}
```
Import `Select, SelectTrigger, SelectValue, SelectContent, SelectItem` from `@/components/ui/select`.

**Step 5 — Pass filteredGroups to TokenGroupTree:**
Find the `<TokenGroupTree groups={masterGroups} ...>` JSX and change it to `<TokenGroupTree groups={filteredGroups} ...>`.

**Important:** The theme selector is only shown when `themes.length > 0`. If a collection has no themes, the Tokens page is completely unchanged.
